CREATE OR REPLACE PROCEDURE "INSERT_N_FUNCIONARIO" (vCOD_FUNCIONARIO number,
vNUM       INT(10)NOT NULL,
  vOPR        CHAR)
IS
  vEXCEPTION EXCEPTION;
BEGIN
  IF (vOPR = 'I') THEN
    INSERT INTO FUNCIONARIO( n_funcionario) VALUES (vNUM);
ELSE
    RAISE vEXCEPTION;
  END IF;
  EXCEPTION
    WHEN vEXCEPTION THEN
      RAISE_APPLICATION_ERROR(-20999,'ATENÇÃO! Operação diferente de I.', FALSE);
END INSERT_N_FUNCIONARIO;
