CREATE OR REPLACE PROCEDURE "CRUD_FUNCIONARIO" (vCOD_FUNCIONARIO number,
  vNOME       VARCHAR2,
  vOPR        CHAR)
IS
  vEXCEPTION EXCEPTION;
BEGIN
  IF (vOPR = 'I') THEN
    INSERT INTO FUNCIONARIO( n_funcionario, n_pessoa, n_gerente, n_agencia ) VALUES (default, vNOME, vNOME, vNOME);
  ELSE
  IF(vOPR = 'A') THEN
    UPDATE FUNCIONARIO SET v_funcionario = vNOME WHERE N_FUNCIONARIO = vCOD_FUNCIONARIO;
  ELSE
  IF(vOPR = 'D')THEN
    DELETE FROM FUNCIONARIO WHERE N_FUNCIONARIO = vCOD_FUNCIONARIO;
  ELSE
    RAISE vEXCEPTION;
  END IF;
  END IF;
  END IF;
  EXCEPTION
    WHEN vEXCEPTION THEN
      RAISE_APPLICATION_ERROR(-20999,'ATENÇÃO! Operação diferente de I, D, A.', FALSE);
END CRUD_FUNCIONARIO;
