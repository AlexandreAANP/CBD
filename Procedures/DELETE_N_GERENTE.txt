CREATE OR REPLACE PROCEDURE "DELETE_N_GERENTE" (vCOD_FUNCIONARIO number,
 vNUM      INT(10),
  vOPR        CHAR)
IS
  vEXCEPTION EXCEPTION;
BEGIN
  IF (vOPR = 'D') THEN
    DELETE FROM FUNCIONARIO WHERE N_GERENTE = vCOD_FUNCIONARIO;
ELSE
    RAISE vEXCEPTION;
  END IF;
  EXCEPTION
    WHEN vEXCEPTION THEN
      RAISE_APPLICATION_ERROR(-20999,'ATENÇÃO! Operação diferente de D.', FALSE);
END DELETE_N_GERENTE;
