CREATE OR REPLACE PROCEDURE "UPDATE_N_PESSOA" (vCOD_FUNCIONARIO number,
 vNUM      INT(20)NOT NULL,
  vOPR        CHAR)
IS
  vEXCEPTION EXCEPTION;
BEGIN
  IF (vOPR = 'U') THEN
    UPDATE FUNCIONARIO SET n_pessoa = vNUM WHERE N_PESSOA = vCOD_FUNCIONARIO;
ELSE
    RAISE vEXCEPTION;
  END IF;
  EXCEPTION
    WHEN vEXCEPTION THEN
      RAISE_APPLICATION_ERROR(-20999,'ATENÇÃO! Operação diferente de U.', FALSE);
END UPDATE_N_PESSOA;
